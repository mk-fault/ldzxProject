## 学生
### 1.获取全部学生信息
```javascript
url : '/api/student/students/'
method : GET
permission : User
params : {
    "page" : 页数,
    "ordering" : 排序方式,

    "id" : 身份证号,
    "sex" : 性别,
    "admission_date" : 入学年份,
    "max_admission_date" : 最大入学年份,
    "min_admission_date" : 最小入学年份,
    "max_student_id" ：最大学号,
    "min_student_id" ：最小学号,
    "student_id" ： 学号,
    "class_num" : 班级,
    "name" : 姓名
}
```

```javascript

### 2.下载通知书
```javascript
url : '/api/offer/download/'
method : POST
permission : Any
data : {
    "id": "513322200003140011",
    "name": "张三",
    "student_id": "18041817"
}
return : .pdf
```
### 3.添加一个通知书
```javascript
url : '/api/offer/offers/'
method : POST
form-data : {
    "background_pic":图片文件,
    "text": 通知书内容
}
```
```json
返回数据示例
{
    "id": 16,
    "background_pic": "http://127.0.0.1:8000/media/offer/background_pic/lab.magiconch.comOne-Last-Image-1680583085903_GmFtLlj.jpg",

    "is_using": true

```
### 4.修改一个通知书
```javascript
url : '/api/offer/offers/<int:id>/'
method : PATCH
form-data or raw-data : {
    "background_pic":图片文件,
    "text": 通知书内容,
    "is_using": 是否启用
}
```
### 5.删除一个通知书
```javascript
url : '/api/offer/offers/<int:id>/'
method : DELETE
```
### 6.通知书预览
```javascript
url : '/api/offer/preview/'
method : GET
permission : User
params : {
    "offer_id" : 通知书ID
}
return : .jpeg
